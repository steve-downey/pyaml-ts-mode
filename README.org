* ~pyaml-ts-mode~
Prog-mode YAML treesitter mode

* Installation
#+begin_src elisp
(use-package pyaml-ts-mode
  :commands (pyaml-ts-mode pyaml-ts-mode-install-grammars pyaml-ts-mode-version)
  :vc (:url "https://github.com/steve-downey/pyaml-ts-mode.git"
            :lisp-dir "lisp"
            :rev :newest)
  :mode ("\\.ya?ml\\'" . pyaml-ts-mode)
  :init
  (add-to-list 'major-mode-remap-alist '(yaml-mode . pyaml-ts-mode))
  (add-to-list 'major-mode-remap-alist '(yaml-ts-mode . pyaml-ts-mode))
  )
#+end_src
